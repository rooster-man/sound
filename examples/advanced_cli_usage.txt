# Advanced CLI Usage Ideas for Your Music App

## 1. Key and Scale Selection
cargo run "key:C|scale:major"
cargo run "key:Fs|scale:minor"  
cargo run "key:D|scale:pentatonic"

## 2. Tempo Control  
cargo run "tempo:slow"    # 150ms per note
cargo run "tempo:medium"  # 250ms per note  
cargo run "tempo:fast"    # 100ms per note

## 3. Note Sequences
cargo run "notes:C,E,G,C|duration:quarter"
cargo run "notes:A,B,C|duration:half|octave:5"

## 4. Multiple Commands
cargo run "key:F|tempo:fast|notes:C,D,E,F,G,A,B,C"

## 5. Chord Progressions
cargo run "chords:C-major,F-major,G-major,C-major"
cargo run "progression:I,vi,IV,V|key:C"

## 6. Musical Patterns
cargo run "pattern:arpeggio|key:G"
cargo run "pattern:scale|direction:up-down|key:D"

## Example Implementation in your main.rs:

match process_args(args) {
    Ok(parts) => {
        for part in parts {
            if part.starts_with("key:") {
                let key_name = &part[4..];
                // Set key for melody
            } else if part.starts_with("tempo:") {
                let tempo = &part[6..];
                // Adjust note durations
            } else if part.starts_with("notes:") {
                let notes = &part[6..];
                // Create custom melody
            }
        }
    }
    Err(_) => {
        // Run default demo
    }
}
